class Flipkart
{
    String products[] = new String[21];
    int index;

    public boolean addFlipkart(String product)
    {
        boolean isFlipkartAdded = false;

        if(product != null && !product.isEmpty())
        {
            products[index++] = product;
            isFlipkartAdded = true;
        }
        else
        {
            System.out.println(product + " is invalid");
        }

        return isFlipkartAdded;
    }

    public void getFlipkart()
    {
        System.out.println("Get Method is Added");

        for(String product : products)
        {
            System.out.println(product);
        }
    }
	public boolean updateFlipkart(String existingFlipkartName,String updateFlipkartName)
	{
		boolean isFlipkartUpdated=false;
		for(int index=0;index<products.length;index++)
		{
			if(products[index] == existingFlipkartName)
			{
				products[index]=updateFlipkartName;
				isFlipkartUpdated=true;
			}
			if(isFlipkartUpdated==false)
			{
				System.out.println(existingFlipkartName + "is not found");
			}
		}
		return isFlipkartUpdated;
	}
	public boolean deleteFlipkart(String value)
	{
		boolean isFlipkartDeleted=false;
		for(int index=0; index<products.length;index++)
		{
			if(products[index]!=null && products[index].equals(value))
			{
				for(int j=index;j<products.length-1;j++)
					products[j]=products[j+1];
					products[products.length-1]=null;
			}
		}
		return isFlipkartDeleted;
	}
}
